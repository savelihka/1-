struct Ресторан {
    let название: String
    let рейтинг: Double
    let времяДоставки: Int
}

struct ЯндексЕда {
    var рестораны: [Ресторан]
    
    mutating func сортироватьПоКритерию(_ criteria: (Ресторан, Ресторан) -> Bool) {
        рестораны.sort(by: criteria)
    }
    
    func отсортироватьПоНазванию() -> [Ресторан] {
        return рестораны.sorted(by: { $0.название < $1.название })
    }
    
    func отсортироватьПоРейтингу() -> [Ресторан] {
        return рестораны.sorted(by: { $0.рейтинг > $1.рейтинг })
    }
    
    func отсортироватьПоВремениДоставки() -> [Ресторан] {
        return рестораны.sorted(by: { $0.времяДоставки < $1.времяДоставки })
    }
}

// Пример использования
var яндексЕда = ЯндексЕда(рестораны: [
    Ресторан(название: "Ресторан A", рейтинг: 4.5, времяДоставки: 30),
    Ресторан(название: "Ресторан B", рейтинг: 4.2, времяДоставки: 25),
    Ресторан(название: "Ресторан C", рейтинг: 4.7, времяДоставки: 20)
])

// Сортировка по рейтингу
яндексЕда.сортироватьПоКритерию({ $0.рейтинг > $1.рейтинг })
print(яндексЕда.рестораны)

// Сортировка по времени доставки
яндексЕда.сортироватьПоКритерию({ $0.времяДоставки < $1.времяДоставки })
print(яндексЕда.рестораны)

// Получение отсортированного списка ресторанов по названию
let отсортированоПоНазванию = яндексЕда.отсортироватьПоНазванию()
print(отсортированоПоНазванию)
